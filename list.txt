! Hide shorts button on sidebar

youtube.com##ytd-mini-guide-entry-renderer:has(span:has-text(/^Shorts$/))

youtube.com##yt-chip-cloud-chip-renderer:has(yt-formatted-string:has-text(/^Shorts$/))

! Hide shorts sections

youtube.com##ytd-rich-section-renderer:has(span:has-text(/^Shorts$/))

youtube.com##yt-tab-shape:has(div:has-text(/^Shorts$/))

youtube.com##div.yt-tab-group-shape-wiz__slider
